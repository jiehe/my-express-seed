一、什么是闭包？

1、官方”的解释是：闭包是一个拥有许多变量和绑定了这些变量的环境的表达式（通常是一个函数），因而这些变量也是该表达式的一部分。
相信很少有人能直接看懂这句话，因为他描述的太学术。其实这句话通俗的来说就是：JavaScript中所有的function都是一个闭包。
2、闭包也可以理解为函数的作用域、上下文。闭包内部可以直接读取上级的局部变量直至全局变量，并且在闭包外部无法读取闭包内的局部变量。

  1. var b = 'baidu';
  2. function outer() {
  3.   var a = 'alibaba';
  4.   var ba = b + "+" + a;
  5.   alert('outer ' + ba);
  6.   //   alert('outer' +bat); 出错
  7.   return function inner() {
  8.     var t = 'tengxun';
  9.     var bat = ba + ' +' + t;
  10.     alert('inner ' + bat);
  11.   }
  12. }
  14. function start() {
  15.   var inner = outer();
  16.   inner();
  17. }

二、闭包的作用

闭包有一个经典的用法。

  1. //html
  2. <ul>
  3.     <li>1</li>
  4.     <li>2</li>
  5.     <li>3</li>
  6.     <li>4</li>
  7.     <li>5</li>
  8. </ul>
  9. //js
  1. var lis = document.getElementsByTagName('li');
  2. for(var i = 0 ,len = lis.length; i<len; i++) {
  3.     lis[i].onclick = function() {
  4.     alert(i);
  5.  }}

上面的例子你会发现每一个li标签点击都是弹出 5.

改成下面就好了

  1. for (var i = 0, len = lis.length; i < len; i++) {
  2.   lis[i].onclick = function(i) {
  3.     return function() {
  4.       alert(i);
  5.     }
  6.   }(i)
  7. }

函数是运行在定义他们的作用域中，而不是运行在调用他们的作用域中。

var fs = require('fs');
var files = ['a.txt', 'b.txt', 'c.txt'];
for (var i = 0; i < files.length; i++) {
(function(i) {
fs.readFile(files[i], 'utf-8', function(err, contents) {
console.log(files[i] + ': ' + contents);
});
})(i);
